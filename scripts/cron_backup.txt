# Cron job configuration for automated database backups
# Add this to your crontab with: crontab -e

# Daily backup at 2 AM
0 2 * * * cd /path/to/ai-attendance-system && python3 scripts/backup_database.py --action backup --suffix daily

# Weekly backup on Sundays at 3 AM (keeps more backups)
0 3 * * 0 cd /path/to/ai-attendance-system && python3 scripts/backup_database.py --action backup --suffix weekly --max-backups 4

# Monthly backup on the 1st at 4 AM
0 4 1 * * cd /path/to/ai-attendance-system && python3 scripts/backup_database.py --action backup --suffix monthly --max-backups 12

# Rotate backups daily to clean up old ones
0 5 * * * cd /path/to/ai-attendance-system && python3 scripts/backup_database.py --action rotate --max-backups 30
